var maxCrashPoint = 10; // set the maximum crash point var cashoutPoint = Math.floor(Math.random() * maxCrashPoint) + 1; // generate a random cashout point function startGame() { // start the game logic var currentMultiplier = 1; var intervalId = setInterval(function() { // update the multiplier logic currentMultiplier += 0.01; if (currentMultiplier >= cashoutPoint) { // cash out automatically clearInterval(intervalId); console.log("Auto cashout at " + cashoutPoint + "x"); } }, 10); }
